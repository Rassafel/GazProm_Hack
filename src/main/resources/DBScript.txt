create table users
(
    id integer primary key
);

create table paths
(
    id           integer REFERENCES users (id),
    timeInterval timestamp,
    lat          float,
    lon          float,
    distance     float
);

create table banks
(
    bankId        integer PRIMARY KEY,
    lat           float,
    lon           float,
    region        text,
    regionType    text,
    setlementType text,
    setlement     text,
    fullAddress   text,
    location      text
);

create table bankStats
(
    bankId       integer REFERENCES banks (bankId),
    timeInterval timestamp,
    countUsers   integer,
    minUsers     integer,
    maxUsers     integer
);


create or replace function calculateDistance(lat1 float, lat2 float, lon1 float, lon2 float) returns float
    language plpgsql as
$$
declare
    distance       float;
    declare tempor double precision;
begin
    select pow(sin(radians(lon2 - lon1)/2), 2) * cos(radians(lat2)) * cos(radians(lat1)) +
           pow(sin(radians(lat2 - lat1)/2), 2)
    into tempor;
    select 2 * atan2(sqrt(tempor), sqrt(1 - tempor)) * 6372795 into distance;
    return distance;
end;
$$;
